# Default server configuration
#
server {

access_log  /data/mysite.local/log/access.log;
    error_log   /data/mysite.local/log/error.log;


root /data/ibuypro/web;

# Add index.php to the list if you are using PHP
index index.php;

server_name ibuypro.1site.space www.ibuypro.1site.space;

location / {
 try_files $uri $uri/ /index.php$is_args$args;
}

# pass PHP scripts to FastCGI server
#
location ~ \.php$ {
                include snippets/fastcgi-php.conf;
        #       # With php7.0-fpm:
                fastcgi_pass unix:/run/php/php7.2-fpm.sock;
        }
location ~ ^/assets/.*\.php$ {
        deny all;
    }

        listen 80;
 listen [::]:80;

}

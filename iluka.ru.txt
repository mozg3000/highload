# Default server configuration
#
server {

access_log  /data/iluka/logs/access.log;
    error_log   /data/iluka/logs/error.log;


root /data/iluka/web;

# Add index.php to the list if you are using PHP
index index.php;

server_name iluka.ru www.iluka.ru;

location / {
 try_files $uri $uri/ /index.php$is_args$args;
}

# pass PHP scripts to FastCGI server
#
location ~ \.php$ {
                include snippets/fastcgi-php.conf;
        #       # With php7.0-fpm:
                fastcgi_pass unix:/run/php/php7.2-fpm.sock;
        }
location ~ ^/assets/.*\.php$ {
        deny all;
    }

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/iluka.ru-0001/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/iluka.ru-0001/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot



}
